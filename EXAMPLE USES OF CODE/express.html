Express.js is a minimal, unopinionated web framework for Node.js used to build servers and APIs.
It sits on top of Node's built-in http module and adds routing, middleware support, and utilities that make building web applications much easier.

Setup:

npm init -y
npm install express
For environment variables (keeping secrets out of code): 
npm install dotenv

Basic Server Example:
const express = require('express');
const app = express();

// Middleware to parse JSON request bodies
app.use(express.json());

// GET route — return plain text
app.get('/', (req, res) => {
  res.send('Hello, World!');
});

// GET with URL parameter
app.get('/users/:id', (req, res) => {
  const { id } = req.params;
  res.json({ userId: id, name: 'Alice' });
});

// POST route — create a new resource
app.post('/users', (req, res) => {
  const { name, email } = req.body;
  res.status(201).json({ message: 'User created', name, email });
});

// PUT route — replace an entire resource
app.put('/users/:id', (req, res) => {
  const { id } = req.params;
  const { name, email } = req.body;
  res.json({ message: `User ${id} updated`, name, email });
});

// DELETE route — remove a resource
app.delete('/users/:id', (req, res) => {
  const { id } = req.params;
  res.status(204).send(); // 204 No Content — success with no body
});

// Start the server
app.listen(3000, () => {
  console.log('Server running on http://localhost:3000');
});

Key Concepts:
express() — Creates an app instance. This is the starting point of every Express application.
app.use() — Registers middleware that runs on every matching request before the route handler.
app.get() / app.post() / app.put() / app.patch() / app.delete() — Registers a route handler for a specific HTTP method and URL path.
req — The incoming request object. Key properties:
  - req.params — URL parameters (e.g. :id)
  - req.body — request body (requires express.json() middleware)
  - req.query — query string (e.g. ?sort=asc)
  - req.headers — request headers (e.g. req.headers.authorization)
  - req.method — HTTP method (GET, POST, etc.)
  - req.path — the URL path
res — The outgoing response object. Key methods:
  - res.send() — send text or HTML
  - res.json() — send JSON (sets Content-Type automatically)
  - res.status(code) — set the HTTP status code (chainable)
  - res.redirect('/path') — redirect the client to another URL
  - res.sendFile(path) — send a file from the filesystem
app.listen(port, callback) — Starts the HTTP server on the given port.

Middleware:
Middleware is any function with the signature (req, res, next) that runs between the request and the route handler.
It can modify req/res, run logic, or end the request. It MUST call next() to pass control to the next step, or the request will hang.
// Built-in middleware:
app.use(express.json());              // parse JSON bodies
app.use(express.urlencoded({ extended: true })); // parse form bodies
app.use(express.static('public'));    // serve static files from the 'public' folder

// Custom middleware — runs on every request:
app.use((req, res, next) => {
  console.log(`${new Date().toISOString()} ${req.method} ${req.path}`);
  next(); // MUST call next() or the request hangs
});

// Route-specific middleware — only runs on this route:
function requireAuth(req, res, next) {
  if (!req.headers.authorization) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  next();
}
app.get('/dashboard', requireAuth, (req, res) => {
  res.send('Welcome to the dashboard!');
});

How a Request Flows:
1. Client sends a request (e.g. POST /users)
2. Global middleware runs in order (e.g. JSON parser, logger)
3. Route-specific middleware runs (e.g. auth check)
4. The route handler runs and sends a response
5. If an error is thrown, Express's error handler middleware catches it

URL Parameters vs Query Strings:
// URL parameter — part of the path; identifies a specific resource:
app.get('/users/:id', (req, res) => {
  console.log(req.params.id); // "42" for /users/42
});

// Multiple parameters:
app.get('/users/:userId/posts/:postId', (req, res) => {
  const { userId, postId } = req.params;
});

// Query string — optional; used for filters, sorting, pagination:
// GET /products?category=shoes&sort=price&limit=10
app.get('/products', (req, res) => {
  const { category, sort, limit = 20 } = req.query;
  res.json({ category, sort, limit });
});

express.Router() — Organising Routes into Separate Files:
As an app grows, putting all routes in one file becomes messy. express.Router() lets you group related routes into their own module.
// ── routes/users.js ──
const router = require('express').Router();

router.get('/', (req, res) => res.json({ message: 'Get all users' }));
router.get('/:id', (req, res) => res.json({ message: `Get user ${req.params.id}` }));
router.post('/', (req, res) => res.status(201).json({ message: 'Create user' }));

module.exports = router;

// ── server.js ──
const usersRouter = require('./routes/users');
app.use('/users', usersRouter); // all routes in usersRouter get '/users' prefix
// → GET /users        hits router.get('/')
// → GET /users/42     hits router.get('/:id')
// → POST /users       hits router.post('/')

Async Route Handlers:
When using async/await in route handlers, you must catch errors and pass them to next().
Otherwise, uncaught Promise rejections will crash the server or hang the request.
// Wrong — unhandled rejection can crash Node.js:
app.get('/data', async (req, res) => {
  const data = await fetchFromDatabase(); // if this throws, the request hangs
  res.json(data);
});

// Correct — wrap in try/catch and call next(err):
app.get('/data', async (req, res, next) => {
  try {
    const data = await fetchFromDatabase();
    res.json(data);
  } catch (err) {
    next(err); // passes the error to the error-handling middleware
  }
});

// Cleaner — a wrapper utility to avoid repeating try/catch everywhere:
function asyncHandler(fn) {
  return (req, res, next) => Promise.resolve(fn(req, res, next)).catch(next);
}

app.get('/data', asyncHandler(async (req, res) => {
  const data = await fetchFromDatabase();
  res.json(data);
}));

Error-Handling Middleware:
Express identifies error-handling middleware by its 4-parameter signature: (err, req, res, next).
It must be registered LAST, after all routes, to catch errors forwarded by next(err).
// Catches any error passed to next(err) anywhere in the app:
app.use((err, req, res, next) => {
  console.error(err.stack);

  const status  = err.status || 500;
  const message = err.message || 'Internal Server Error';

  res.status(status).json({ error: message });
});

// 404 handler — put BEFORE the error handler but AFTER all routes:
app.use((req, res) => {
  res.status(404).json({ error: 'Route not found' });
});

Environment Variables (.env):
Never hard-code secrets (database passwords, API keys) into your code.
Store them in a .env file and add that file to .gitignore.
# .env file (NEVER commit this to git)
PORT=3000
DB_URL=mongodb://localhost:27017/myapp
JWT_SECRET=supersecretkey

// server.js
require('dotenv').config(); // loads .env into process.env

const PORT   = process.env.PORT || 3000;
const DB_URL = process.env.DB_URL;

app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});

Serving Static Files:
// Serve everything in the 'public' folder at the root URL:
app.use(express.static('public'));
// → public/index.html  available at /
// → public/style.css   available at /style.css
// → public/img/logo.png available at /img/logo.png

// With a URL prefix:
app.use('/static', express.static('public'));
// → public/logo.png available at /static/logo.png

HTTP Status Codes in Express:
res.status(200).json({}) — OK (default for successful GET)
res.status(201).json({}) — Created (use after POST creates a resource)
res.status(204).send() — No Content (use after DELETE — no body)
res.status(400).json({}) — Bad Request (invalid input from client)
res.status(401).json({}) — Unauthorized (not logged in)
res.status(403).json({}) — Forbidden (logged in but no permission)
res.status(404).json({}) — Not Found (resource doesn't exist)
res.status(500).json({}) — Internal Server Error (server crashed)

Testing Routes:
Test your Express routes from the terminal without a browser:
# GET request
curl http://localhost:3000/users/42

# POST request with JSON body
curl -X POST http://localhost:3000/users \
  -H "Content-Type: application/json" \
  -d '{"name":"Bob","email":"bob@example.com"}'

# PUT request
curl -X PUT http://localhost:3000/users/42 \
  -H "Content-Type: application/json" \
  -d '{"name":"Robert","email":"robert@example.com"}'

# DELETE request
curl -X DELETE http://localhost:3000/users/42
Alternatively, use tools like Postman, Insomnia, or the Thunder Client VS Code extension.
